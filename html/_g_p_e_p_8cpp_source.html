263 void GPEP::gausshermite(int n, std::shared_ptr<IMatrix> x, std::shared_ptr<IMatrix> w)
264 {
265  std::shared_ptr<IMatrix> x0 = algebra->createZeros(x->getLength(), 1);
266  std::shared_ptr<IMatrix> w0 = algebra->createZeros(w->getLength(), 1);
267  int m = (n + 1) / 2;
268  double z = 0, pp = 0, p1 = 0, p2 = 0, p3 = 0;
269  for (int i = 0; i < m; i++) {
270  if (i == 0)
271  z = sqrt(2.0 * (double)n + 1.0) - 1.85575
272  * pow(2.0 * (double)n + 1.0, -0.16667);
273  else if (i == 1)
274  z = z - 1.14 * pow((double)n, 0.426) / z;
275  else if (i == 2)
276  z = 1.86 * z - 0.86 * x0->get(0);
277  else if (i == 3)
278  z = 1.91 * z - 0.91 * x0->get(1);
279  else
280  z = 2.0 * z - x0->get(i - 2);
281 
282  for (int its = 0; its < 10; its++) {
283  p1 = 1.0 / sqrt(sqrt(M_PI));
284  p2 = 0.0;
285  for (double j = 1; j <= n; j++) {
286  p3 = p2;
287  p2 = p1;
288  const double a = z * sqrt(2.0 / (double)j) * p2;
289  const double b = sqrt((j - 1) / j) * p3;
290  p1 = a - b;
291 // System.currentTimeMillis();
292  }
293  pp = sqrt(2.0 * (double)n) * p2;
294  double z1 = z;
295  z = z1 - p1 / pp;
296  if (std::abs(z - z1) < 2.2204e-16)
297  break;
298  }
299 
300  x0->put(i, z);
301  x0->put(n - 1 - i, -z); // x(n+1-i) = -z;
302  w0->put(i, 2 / (pp * pp)); // w(i) = 2/(pp*pp);
303  w0->put(n - 1 - i, w0->get(i)); // w(n+1-i) = w(i);
304  }
305 
306  w0 = w0->div(sqrt(M_PI)); // w = w/sqrt(pi);
307  x0 = x0->mul(sqrt(2.0));
308  x0 = x0->sort(); // x = sort(x);
309  x->copy(x0);
310  w->copy(w0);
311 }
