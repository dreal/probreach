15 bool stability::jury_test(std::vector<double> poly)
16 {
17  // getting the size of the polynomials
18  size_t n = poly.size()-1;
19 
20  // condition 1
21  if(abs(poly.back()) >= poly.front()) return false;
22 
23  // condition 2
24  double res = 0;
25  for(double value : poly)
26  {
27  res += value;
28  }
29  if(res <= 0) return false;
30 
31  // condition 3
32  // n is even
33  if(n % 2 == 0)
34  {
35  res = 0;
36  double sign = 1;
37  for(double value : poly)
38  {
39  res += sign*value;
40  sign = -sign;
41  }
42  if(res <= 0) return false;
43  }
44  // n is odd
45  else
46  {
47  res = 0;
48  double sign = -1;
49  for(double value : poly)
50  {
51  res += sign*value;
52  sign = -sign;
53  }
54  if(res >= 0) return false;
55  }
56 
57  // condition 4
58  // reversing the table
59  for(size_t j = 0; j < n - 2; j++)
60  {
61  size_t m = poly.size() - 1;
62  vector<double> new_row;
63  for(size_t i = 0; i < m; i++)
64  {
65  new_row.push_back(poly.at(m)*poly.at(i+1) - poly.at(0)*poly.at(m-1-i));
66  }
67  //reverse(new_row.begin(), new_row.end());
68  if(abs(new_row.back()) <= abs(new_row.front())) return false;
69  //reverse(new_row.begin(), new_row.end());
70  poly = new_row;
71  }
72 
73  return true;
74 }
