65 capd::interval algorithm::evaluate_pha(int min_depth, int max_depth)
66 {
67  CLOG_IF(global_config.verbose, INFO, "algorithm")
68  << "Obtaining partition of domain of continuous random parameters";
69  // getting partition of domain of continuous random variables
70  std::vector<box> init_rv_partition = measure::get_rv_partition();
71  // getting domain of continuous random variables
72  box rv_domain = measure::bounds::get_rv_domain();
73  // here we start with entire domain instead of partition
74  if (!global_config.partition_prob) {
75  init_rv_partition.clear();
76  init_rv_partition.push_back(rv_domain);
77  }
78  // getting partition of domain of discrete random variables
79  std::vector<box> dd_partition = measure::get_dd_partition();
80  if (dd_partition.empty()) {
81  dd_partition.push_back(box());
82  }
83  // checking if there are multiple initial and goal states
84  if ((pdrh::init.size() > 1) || (pdrh::goal.size() > 1)) {
85  CLOG(WARNING, "algorithm") << "Multiple initial or goal states are not supported";
86  }
87  capd::interval probability(0, 1);
88  // checking if there are any continuous random variables
89  CLOG_IF(global_config.verbose, INFO, "algorithm") << "P = " << probability;
90  // generating all paths of lengths [min_depth, max_depth]
91  std::vector<std::vector<pdrh::mode *>> paths;
92  for (int i = min_depth; i <= max_depth; i++) {
93  std::vector<std::vector<pdrh::mode *>> paths_i = pdrh::get_paths(pdrh::get_mode(pdrh::init.front().id),
94  pdrh::get_mode(pdrh::goal.front().id), i);
95  paths.insert(paths.cend(), paths_i.cbegin(), paths_i.cend());
96  }
97  //resulting probability
98  capd::interval res_prob(0.0);
99  // evaluating boxes
100  //#pragma omp parallel for
101  for (box dd : dd_partition) {
102  if (pdrh::rv_map.size() > 0) {
103  probability = capd::interval(0, 2 - measure::p_measure(rv_domain).leftBound());
104  }
105  vector<box> rv_partition = init_rv_partition;
106  std::vector<box> rv_stack;
107  while (capd::intervals::width(probability) > global_config.precision_prob) {
108  //for(box rv : rv_partition)
109  //#pragma omp parallel for
110  for (unsigned long i = 0; i < rv_partition.size(); i++) {
111  box rv = rv_partition.at(i);
112  // calculating probability measure of the box
113  // initially p_box = [1.0, 1.0]
114  CLOG_IF(global_config.verbose, INFO, "algorithm") << "====================";
115  capd::interval p_box(1);
116  if (!dd.empty()) {
117  p_box *= measure::p_dd_measure(dd);
118  CLOG_IF(global_config.verbose, INFO, "algorithm") << "dd_box: " << dd;
119  }
120  if (!rv.empty()) {
121  p_box *= measure::p_measure(rv);
122  CLOG_IF(global_config.verbose, INFO, "algorithm") << "rv_box: " << rv;
123  }
124  CLOG_IF(global_config.verbose, INFO, "algorithm") << "p_box: " << p_box;
125  // evaluating boxes
126  std::vector<box> boxes{dd, rv};
127  // undetermined answers counter
128  int undet_counter = 0;
129  // solver timeout counter
130  int timeout_counter = 0;
131  // unsat counter
132  int unsat_counter = 0;
133  // sat flag
134  bool sat_flag = false;
135  // evaluating all paths for all dd and rv
136  //cout << "Before evaluate loop " << omp_get_thread_num() << endl;
137  for (std::vector<pdrh::mode *> path : paths) {
138  std::string solver_opt;
139  std::stringstream p_stream;
140  for (pdrh::mode *m : path) {
141  p_stream << m->id << " ";
142  }
143  // removing trailing whitespace
144  CLOG_IF(global_config.verbose, INFO, "algorithm") << "Path: " << p_stream.str().substr(0,
145  p_stream.str().find_last_of(
146  " "));
147  // changing solver precision
148  solver_opt = global_config.solver_opt;
149  std::stringstream s;
150  s << solver_opt << " --precision " <<
151  measure::volume(rv).leftBound() * global_config.solver_precision_ratio;
152  global_config.solver_opt = s.str();
153  int res = decision_procedure::evaluate(path, boxes);
154  // setting old precision
155 #pragma omp critical
156  {
157  global_config.solver_opt = solver_opt;
158  switch (res) {
159  case decision_procedure::SAT:
160  if (p_box.leftBound() > 0) {
161  probability = capd::interval(probability.leftBound() + p_box.leftBound(),
162  probability.rightBound());
163  }
164  CLOG_IF(global_config.verbose, INFO, "algorithm") << "SAT";
165  CLOG_IF(global_config.verbose, INFO, "algorithm") << "P = " << probability;
166  /*
167  if(capd::intervals::width(probability) <= global_config.precision_prob)
168  {
169  return probability;
170  }
171  */
172  sat_flag = true;
173  break;
174 
175  case decision_procedure::UNSAT:
176  CLOG_IF(global_config.verbose, INFO, "algorithm") << "UNSAT";
177  unsat_counter++;
178  break;
179  case decision_procedure::UNDET:
180  CLOG_IF(global_config.verbose, INFO, "algorithm") << "UNDEC";
181  undet_counter++;
182  break;
183  case decision_procedure::ERROR:
184  CLOG(ERROR, "algorithm") << "Error occurred while calling the solver";
185  exit(EXIT_FAILURE);
186  break;
187  case decision_procedure::SOLVER_TIMEOUT:
188  CLOG_IF(global_config.verbose, INFO, "algorithm") << "SOLVER_TIMEOUT";
189  timeout_counter++;
190  break;
191  default:
192  break;
193  }
194  }
195  // breaking out of the loop if a sat path was found
196  if (sat_flag) {
197  break;
198  }
199  }
200  // checking if there are no sat answers
201  if (!sat_flag) {
202  // if the box is undetermined on either path
203  if ((undet_counter > 0) || (timeout_counter > 0)) {
204  CLOG_IF(global_config.verbose, INFO, "algorithm") << "Bisect " << rv;
205  std::vector<box> rv_bisect = box_factory::bisect(rv);
206  rv_stack.insert(rv_stack.end(), rv_bisect.begin(), rv_bisect.end());
207  }
208  // if the box is unsat for all paths
209  else if (unsat_counter == paths.size()) {
210  if (p_box.rightBound() > 0) {
211  probability = capd::interval(probability.leftBound(),
212  probability.rightBound() - p_box.leftBound());
213  }
214  CLOG_IF(global_config.verbose, INFO, "algorithm") << "P = " << probability;
215  /*
216  if (capd::intervals::width(probability) <= global_config.precision_prob)
217  {
218  return probability;
219  }
220  */
221  }
222  }
223  //cout << "P = " << probability << endl;
224  }
225  // copying the bisected boxes from the stack to partition
226  rv_partition = rv_stack;
227  rv_stack.clear();
228  // breaking out of the loop if there are no continuous random variables
229  if (pdrh::rv_map.size() == 0) {
230  rv_partition.push_back(box());
231  break;
232  }
233  }
234  if (!dd.empty()) {
235  capd::interval dd_measure = measure::p_dd_measure(dd);
236  res_prob += probability * dd_measure;
237  CLOG_IF(global_config.verbose, INFO, "algorithm") << "P(" << dd << ") = " << probability * dd_measure;
238  } else {
239  res_prob = probability;
240  }
241 
242  }
243  return res_prob;
244 }
