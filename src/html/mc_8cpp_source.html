26 capd::interval algorithm::evaluate_pha_chernoff(int min_depth, int max_depth, double acc, double conf, vector<box> nondet_boxes) {

32  r = gsl_rng_alloc(T);
33  // setting the seed

35  // getting sample size with recalculated confidence
36  long int sample_size = algorithm::get_cernoff_bound(acc, std::sqrt(conf));
37  long int sat = 0;
38  long int unsat = 0;
39  CLOG_IF(global_config.verbose_result, INFO, "algorithm") << "Chernoff-Hoeffding algorithm started";
40  CLOG_IF(global_config.verbose_result, INFO, "algorithm") << "Random sample size: " << sample_size;
41 #pragma omp parallel for schedule (dynamic)
42  for (long int ctr = 0; ctr < sample_size; ctr++) {
43  std::vector<std::vector<pdrh::mode *>> paths;
44  // getting all paths
45  for (pdrh::state i : pdrh::init) {
46  for (pdrh::state g : pdrh::goal) {

51  }
52  }
53  }

56  CLOG_IF(global_config.verbose, INFO, "algorithm") << "Random sample: " << b;

58  boxes.insert(boxes.end(), nondet_boxes.begin(), nondet_boxes.end());
59  int undet_counter = 0;
60  int timeout_counter = 0;
61  bool sat_flag = false;
62  // evaluating all paths
63  for (std::vector<pdrh::mode *> path : paths) {
64  std::stringstream p_stream;
65  for (pdrh::mode *m : path) {
66  p_stream << m->id << " ";
67  }
68  // removing trailing whitespace
69  CLOG_IF(global_config.verbose, INFO, "algorithm") << "Path: " << p_stream.str().substr(0,
70  p_stream.str().find_last_of(
71  " "));

74  res = decision_procedure::evaluate_delta_sat(path, boxes);
75  } else {
76  res = decision_procedure::evaluate(path, boxes);
77  }

79  {
80  if (res == decision_procedure::SAT) {
81  CLOG_IF(global_config.verbose, INFO, "algorithm") << "SAT";

97  }
98  }
99  if (sat_flag) {
100  break;
101  }
102  }
103  // updating unsat counter
104 #pragma omp critical
105  {
106  if ((undet_counter == 0) && (timeout_counter == 0) && (!sat_flag)) {
107  unsat++;
108  }
109  CLOG_IF(global_config.verbose, INFO, "algorithm") << "CI: " << capd::interval(
110  ((double) sat / (double) sample_size) - acc,
111  ((double) (sample_size - unsat) / (double) sample_size) + acc);
112  CLOG_IF(global_config.verbose, INFO, "algorithm") << "Progress: " << (double) ctr / (double) sample_size;
113  }
114  }
115  gsl_rng_free(r);
116  CLOG_IF(global_config.verbose_result, INFO, "algorithm") << "Chernoff-Hoeffding algorithm finished";
117  return capd::interval(((double) sat / (double) sample_size) - acc,
118  ((double) (sample_size - unsat) / (double) sample_size) + acc);

276 long int algorithm::get_cernoff_bound(double acc, double conf) {
277  if ((acc <= 0) || (conf < 0) || (conf >= 1)) {
278  CLOG(ERROR, "algorithm") << "accuracy must be descending than 0 and confidence must be inside [0, 1) interval";
279  }
280  return (long int) std::ceil((1 / (2 * acc * acc)) * std::log(2 / (1 - conf)));

537 pair<capd::interval, box> algorithm::solve_min_max() {
538  const gsl_rng_type *T;
539  gsl_rng *r;
540  gsl_rng_env_setup();
541  T = gsl_rng_default;
542  // creating random generator
543  r = gsl_rng_alloc(T);
544  // setting the seed
545  gsl_rng_set(r, std::chrono::system_clock::now().time_since_epoch() / std::chrono::milliseconds(1));
546 
547  // the first element is the value of the objective function
548  // the second element is the values of the nondet parameters
549  std::map<capd::interval, box> rob_map;
550 
551  box nondet_domain = pdrh2box::get_nondet_domain();
552  //initializing probability value
553  box mean = nondet_domain.get_mean();
554  box sigma = nondet_domain.get_stddev();
555  box var = sigma * sigma;
556 
557  unsigned long sample_size = (unsigned long) ceil(
558  global_config.sample_size / measure::get_sample_prob(nondet_domain, mean, sigma).rightBound());
559 
560  pair<capd::interval, box> res = make_pair(capd::interval(-numeric_limits<double>::max()), box());
561 
562 #pragma omp parallel for
563  for (int i = 0; i < sample_size; i++) {
564  // taking a sample form the nondeterministic parameter space
565  box nondet_box = rnd::get_normal_random_sample(r, mean, sigma);
566  if (nondet_domain.contains(nondet_box)) {
567  capd::interval rob(0);
568  for (int j = 0; j < global_config.sample_size; j++) {
569  //cout << "i = " << i << " j = " << j << endl;
570  box random_box = rnd::get_random_sample(r);
571  vector<vector<pdrh::mode *>> paths = ap::get_all_paths({nondet_box, random_box});
572  // computing an objective function over a set of paths as an average value
573  for (vector<pdrh::mode *> path : paths) {
574  capd::interval sample_rob = ap::compute_robustness(path, ap::init_to_box({nondet_box, random_box}),
575  {nondet_box, random_box}) /
576  (double) paths.size();
577 # pragma omp critical
578  {
579  rob = rob + sample_rob;
580  }
581  }
582  }
583 # pragma omp critical
584  {
585  // computing the statistical mean of the objective function
586  rob = rob / global_config.sample_size;
587  // updating the result here
588  if (rob.leftBound() > res.first.rightBound()) {
589  res = make_pair(rob, nondet_box);
590  }
591  // adding to the list of objective functions
592  rob_map.insert(make_pair(rob, nondet_box));
593  }
594  }
595  }

600  }
