175 vector<int> model::find_shortest_path()
176 {
177  // initializing the set of paths
178  vector<vector<int>> paths;
179  vector<int> path;
180  for (pair<int, node> i : this->inits)
181  {
182  for (pair<int, node> g : this->goals)
183  {
184  if (i.first == g.first)
185  {
186  return vector<int>{ i.first };
187  }
188  else
189  {
190  // pushing the initial mode to the initial path
191  path.push_back(i.first);
192  // pushing the initial path to the set of paths
193  paths.push_back(path);
194  while (!paths.empty())
195  {
196  // getting the first path in the set of paths
197  path = paths.front();
198  paths.erase(paths.begin());
199  // getting the mode in the path
200  int mode_id = path.back();
201  vector<int> successors = get_mode(mode_id).get_successors();
202  // proceeding if the current mode has successors
203  if(!successors.empty())
204  {
205  // checking if one of the successors is the end mode
206  if (find(successors.begin(), successors.end(), g.first) != successors.end())
207  {
208  path.push_back(g.first);
209  paths.clear();
210  return path;
211  }
212  else
213  {
214  // iterating through the successors of the current mode
215  for (int s : successors)
216  {
217  // checking if a successor does not appear in the current path
218  if (find(path.begin(), path.end(), s) == path.end())
219  {
220  vector<int> tmp_path = path;
221  tmp_path.push_back(s);
222  paths.push_back(tmp_path);
223  }
224  }
225  }
226  }
227  }
228  }
229  }
230  }
231  path.clear();
232  return path;
233 }

235 vector<vector<int>> model::find_all_paths_of_length(int length)
236 {
237  // initializing the set of paths
238  vector<vector<int>> paths;
239  for(pair<int, node> i : this->inits)
240  {
241  for(pair<int, node> g : this->goals)
242  {
243  vector<int> path;
244  path.push_back(i.first);
245  // initializing the stack
246  vector<vector<int>> stack;
247  stack.push_back(path);
248  while(!stack.empty())
249  {
250  // getting the first paths from the set of paths
251  path = stack.front();
252  stack.erase(stack.cbegin());
253  // checking if the correct path of the required length is found
254  if((path.back() == g.first) && (path.size() == length + 1))
255  {
256  // checking if the path already exists
257  if(find(paths.begin(), paths.end(), path) == paths.end())
258  {
259  paths.push_back(path);
260  }
261  }
262  // proceeding only if the length of the current path is ascending then the required length
263  else if(path.size() < length + 1)
264  {
265  // getting the last mode in the path
266  int mode_id = path.back();
267  // getting the successors of the mode
268  vector<int> successors = get_mode(mode_id).get_successors();
269  for(int s : successors)
270  {
271  // appending the successor the current paths
272  vector<int> new_path = path;
273  new_path.push_back(s);
274  // pushing the new path to the set of the paths
275  stack.push_back(new_path);
276  }
277  }
278  }
279  }
280  }
281  return paths;
