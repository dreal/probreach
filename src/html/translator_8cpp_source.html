355 string translator::Translator::translate_jump_guard(pdrh::node *guard, int mode_id)
356 {
357  stringstream s, value;
358  if (pdrh::var_map.count(guard->value)){
359  value << slStateBase << mode_id << "." << guard->value;
360  } else if (guard->value == "="){
361  value << "==";
362  } else if (guard->value == "and"){
363  if (guard->operands.size() == 1){
364  translate_jump_guard(guard->operands.front(), mode_id);
365  } else {
366  value << " & ";
367  }
368  } else if (guard->value == "or"){
369  value << " | ";
370  }
371  else {
372  value << guard->value;
373  }
374  // checking whether it is an operation node
375  if(guard->operands.size() > 1)
376 
377  {
378  s << "(";
379  for(int i = 0; i < guard->operands.size() - 1; i++)
380  {
381  s << translate_jump_guard(guard->operands.at(i), mode_id);
382  s << value.str();
383 
384  }
385  s << translate_jump_guard(guard->operands.at(guard->operands.size() - 1), mode_id) << ")";
386  }
387  else if(guard->operands.size() == 1)
388  {
389  if(value.str() == "-")
390  {
391  s << "(" << value.str() << translate_jump_guard(guard->operands.front(), mode_id) << ")";
392  }
393  else
394  {
395  s << value.str() << "(" << translate_jump_guard(guard->operands.front(), mode_id) << ")";
396  }
397  }
398  else
399  {
400  s << value.str();
401  }
402  return s.str();
403 }

506  }

515  }
516  }
517

522  for (pdrh::mode m : pdrh::modes){
523  CLOG_IF(global_config.verbose, INFO, "translator") << "Transitions for mode " << m.id;
524  add_state_transition(m);
525  }
526

528  CLOG_IF(global_config.verbose, INFO, "translator") << "Setting simulation time to " << pdrh::time.second->value;
529  this->engine->eval(convertUTF8StringToUTF16String("set_param(mdlName, 'StopTime', '" + pdrh::time.second->value +"');"));
530  } else {
531  CLOG(INFO, "translator") << "Simulation time not set, using default value of 10.";
532  }

535  CLOG(INFO, "translator" ) << "Completed translating model";
