248 // getting the shortest path between two modes
249 vector<pdrh::mode*> pdrh::get_shortest_path(pdrh::mode* begin, pdrh::mode* end)
250 {
251  // initializing the set of paths
252  vector<std::vector<pdrh::mode*>> paths;
253  vector<pdrh::mode*> path;
254  // checking if the initial state is the end state
255  if(begin == end)
256  {
257  // pushing the initial mode to the initial path
258  path.push_back(begin);
259  return path;
260  }
261  else
262  {
263  // pushing the initial mode to the initial path
264  path.push_back(begin);
265  // pushing the initial path to the set of paths
266  paths.push_back(path);
267  while(!paths.empty())
268  {
269  // getting the first path in the set of paths
270  path = paths.front();
271  paths.erase(paths.cbegin());
272  // getting the mode in the path
273  pdrh::mode* cur_mode = path.back();
274  vector<pdrh::mode*> successors = pdrh::get_successors(cur_mode);
275  // proceeding if the current mode has successors
276  if(!successors.empty())
277  {
278  // checking if one of the successors is the end mode
279  if (std::find(successors.cbegin(), successors.cend(), end) != successors.cend())
280  {
281  path.push_back(end);
282  paths.clear();
283  return path;
284  }
285  else
286  {
287  // iterating through the successors of the current mode
288  for (pdrh::mode *suc_mode : successors)
289  {
290  // checking if a successor does not appear in the current path
291  if (find(path.cbegin(), path.cend(), suc_mode) == path.cend())
292  {
293  vector<pdrh::mode*> tmp_path = path;
294  tmp_path.push_back(suc_mode);
295  paths.push_back(tmp_path);
296  }
297  }
298  }
299  }
300  }
301  }
302  path.clear();
303  return path;
304 }

306 // getting all paths of length path_length between begin and end modes
307 vector<vector<pdrh::mode*>> pdrh::get_paths(pdrh::mode* begin, pdrh::mode* end, int path_length)
308 {
309  // initializing the set of paths
310  vector<std::vector<pdrh::mode*>> paths;
311  vector<pdrh::mode*> path;
312  path.push_back(begin);
313  // initializing the stack
314  vector<vector<pdrh::mode*>> stack;
315  stack.push_back(path);
316  while(!stack.empty())
317  {
318  // getting the first paths from the set of paths
319  path = stack.front();
320  stack.erase(stack.cbegin());
321  // checking if the correct path of the required length is found
322  if((path.back() == end) && (path.size() == path_length + 1))
323  {
324  paths.push_back(path);
325  }
326  // proceeding only if the length of the current path is ascending then the required length
327  else if(path.size() < path_length + 1)
328  {
329  // getting the last mode in the path
330  pdrh::mode* cur_mode = path.back();
331  // getting the successors of the mode
332  vector<pdrh::mode*> successors = pdrh::get_successors(cur_mode);
333  for(pdrh::mode* suc_mode : successors)
334  {
335  // appending the successor the current paths
336  vector<pdrh::mode*> new_path = path;
337  new_path.push_back(suc_mode);
338  // pushing the new path to the set of the paths
339  stack.push_back(new_path);
340  }
341  }
342  }
343  return paths;
344 }
