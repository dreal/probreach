119  // returning a value only if the variable from the node appears in the vector of values

122  return vals[n->value];

124  // in case of infinity
125  else if(n->value == "-infty")

127  return -numeric_limits<double>::max();

129  else if(n->value == "infty")

131  return numeric_limits<double>::max();
132  }
133  // in case of a constant
134  double val;
